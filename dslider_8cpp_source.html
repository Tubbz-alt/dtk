364 void DSlider::mousePressEvent(QMouseEvent *event)
365 {
366  QAbstractSlider::mousePressEvent(event);
367 
368  if (event->button() == Qt::LeftButton) {
369  if (orientation() == Qt::Vertical) {
370  setValue(minimum() + ((maximum() - minimum()) * (height() - event->y())) / height()) ;
371  } else {
372  // FIXME
373  // the value 10 is specified in DSlider.theme, it's ugly here, but I don't have any
374  // good idea for now, maybe someone can help.
375  setValue(minimum() + ((maximum() - minimum()) * (event->x() - 10)) / (width() - 10 - 10)) ;
376  }
377 
378  event->accept();
379 
380  Q_D(DSlider);
381 
382  QStyleOptionSlider opt;
383  initStyleOption(&opt);
384  setRepeatAction(SliderNoAction);
385  QRect sr = style()->subControlRect(QStyle::CC_Slider, &opt, QStyle::SC_SliderHandle, this);
386 
387  d->clickOffset = d->pick(event->pos() - sr.topLeft());
388  d->mousePressed = true;
389 
390  Q_EMIT sliderPressed();
391  }
392 }

405 void DSlider::paintEvent(QPaintEvent *event)
406 {
407  Q_D(DSlider);
408 
409  QPainter painter;
410  painter.begin(this);
411 
412  // draw tips
413  QFont font = painter.font();
414  font.setPixelSize(12);
415  painter.setFont(font);
416 
417  QPen pen = painter.pen();
418  pen.setColor(d->m_tipColor);
419  painter.setPen(pen);
420 
421  QRect tmp = rect().adjusted(CustomDrawingLeftPadding - 5, 0, -CustomDrawingRightPadding + 5, 0);
422 
423  QTextOption leftBottomOption;
424  leftBottomOption.setAlignment(Qt::AlignLeft | Qt::AlignBottom);
425  painter.drawText(tmp, d->m_leftTip, leftBottomOption);
426 
427  QTextOption rightBottomOption;
428  rightBottomOption.setAlignment(Qt::AlignRight | Qt::AlignBottom);
429  painter.drawText(tmp, d->m_rightTip, rightBottomOption);
430 
431  // draw scales
432  pen.setColor(d->m_scaleColor);
433  painter.setPen(pen);
434 
435  Q_FOREACH(int scale, d->m_scales) {
436  int x = d->getScalePosition(scale);
437  int y = height() - 8;
438  painter.drawLine(x, y, x, y - CustomDrawingScaleHeight);
439  }
440 
441  if (d->m_handleHovering && !d->m_hoverTimout) {
442  QString str = QString::number(value());
443  int x = d->getScalePosition(value()) - painter.fontMetrics().width(str) / 2.0;
444  painter.setPen(d->m_hoverValueColor);
445  painter.drawText(x, 10, str);
446  }
447 
448  painter.end();
449 
450  QSlider::paintEvent(event);
451 }

453 void DSlider::mouseMoveEvent(QMouseEvent *event)
454 {
455  Q_D(DSlider);
456 
457  if(d->mousePressed) {
458  int newPosition = d->pixelPosToRangeValue(d->pick(event->pos()) - d->clickOffset);
459  setSliderPosition(newPosition);
460 
461  Q_EMIT sliderMoved(newPosition);
462  }
463 
464  if (!d->m_hoverShowValue) {
465  return;
466  }
467 
468  QPoint pos = event->pos();
469  QRect rect(d->getScalePosition(value()) - CustomDrawingLeftPadding, 10, 20, 20);
470  if (d->m_handleHovering) {
471  d->m_handleHovering = rect.contains(pos);
472  } else {
473  d->m_handleHovering = rect.contains(pos);
474  if (d->m_handleHovering) {
475  d->m_hoverTimout = false;
476  if (d->m_hoverShowValueInterval > 0) {
477  d->m_hoverTimer.start(d->m_hoverShowValueInterval);
478  }
479  }
480  }
481 
482  update();
483 }

618 int DSliderPrivate::getScalePosition(int value)
619 {
620  Q_Q(DSlider);
621 
622  float valueRange = q->maximum() - q->minimum();
623  float posRange = q->width() - CustomDrawingLeftPadding - CustomDrawingRightPadding;
624  return CustomDrawingLeftPadding + (value - q->minimum()) * posRange / valueRange;
625 }

627 int DSliderPrivate::pixelPosToRangeValue(int pos) const
628 {
629  Q_Q(const DSlider);
630 
631  QStyleOptionSlider opt;
632  q->initStyleOption(&opt);
633  QRect gr = q->style()->subControlRect(QStyle::CC_Slider, &opt, QStyle::SC_SliderGroove, q);
634  QRect sr = q->style()->subControlRect(QStyle::CC_Slider, &opt, QStyle::SC_SliderHandle, q);
635  int sliderMin, sliderMax, sliderLength;
636 
637  if (q->orientation() == Qt::Horizontal) {
638  sliderLength = sr.width();
639  sliderMin = gr.x();
640  sliderMax = gr.right() - sliderLength + 1;
641  } else {
642  sliderLength = sr.height();
643  sliderMin = gr.y();
644  sliderMax = gr.bottom() - sliderLength + 1;
645  }
646  return QStyle::sliderValueFromPosition(q->minimum(), q->maximum(), pos - sliderMin,
647  sliderMax - sliderMin, opt.upsideDown);
648 }
