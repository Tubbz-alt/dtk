83 void DScrollArea::setAutoHideScrollBar(bool autoHideScrollBar)
84 {
85  Q_D(DScrollArea);
86 
87  if (d->autoHideScrollBar == autoHideScrollBar)
88  return;
89 
90  d->autoHideScrollBar = autoHideScrollBar;
91 
92  if(autoHideScrollBar){
93  if(!d->scrollBarAnimation)
94  d->scrollBarAnimation = new QPropertyAnimation(this, "windowOpacity");
95  if(!d->timer)
96  d->timer = new QTimer(this);
97 
98  d->scrollBarAnimation->setDuration(300);
99  d->timer->setSingleShot(true);
100 
101  d->vBar->hide();
102  d->hBar->hide();
103 
104  connect(d->timer, &QTimer::timeout, [&]{
105  if(d->vBar->isVisible())
106  hideScrollBar(d->vBar);
107  if(d->hBar->isVisible())
108  hideScrollBar(d->hBar);
109  });
110  }else{
111  if(d->scrollBarAnimation)
112  d->scrollBarAnimation->deleteLater();
113  if(d->timer)
114  d->timer->deleteLater();
115 
116  showScrollBar(d->vBar);
117  showScrollBar(d->hBar);
118  }
119 
120  Q_EMIT autoHideScrollBarChanged(autoHideScrollBar);
121 }

142 void DScrollArea::showScrollBar(QScrollBar *bar)
143 {
144  Q_D(DScrollArea);
145 
146  if(bar->isVisible())
147  return;
148 
149  if(bar == d->vBar && verticalScrollBarPolicy() != Qt::ScrollBarAlwaysOff)
150  d->vBar->show();
151  else if(bar == d->hBar && horizontalScrollBarPolicy() != Qt::ScrollBarAlwaysOff)
152  d->hBar->show();
153 
154  if(d->scrollBarAnimation){
155  disconnect(d->scrollBarAnimation, &QPropertyAnimation::finished, bar, &DScrollBar::hide);
156  disconnect(d->scrollBarAnimation, &QPropertyAnimation::finished, bar, &DScrollBar::hide);
157 
158  d->scrollBarAnimation->setStartValue(1.0);
159  d->scrollBarAnimation->setEndValue(0.0);
160  d->scrollBarAnimation->start();
161  d->timer->start(1000);
162  }
