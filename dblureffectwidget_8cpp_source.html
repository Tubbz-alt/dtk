113 bool DBlurEffectWidgetPrivate::updateWindowBlurArea(QWidget *topLevelWidget)
114 {
115  if (!topLevelWidget->isVisible()) {
116  return false;
117  }
118 
119  QList<const DBlurEffectWidget *> blurEffectWidgetList = blurEffectWidgetHash.values(topLevelWidget);
120 
121  bool isExistMaskPath = false;
122 
123  Q_FOREACH (const DBlurEffectWidget *w, blurEffectWidgetList) {
124  if (!w->d_func()->maskPath.isEmpty() && w->isVisible()) {
125  isExistMaskPath = true;
126  break;
127  }
128  }
129 
130  bool ok = false;
131 
132  if (isExistMaskPath) {
133  QList<QPainterPath> pathList;
134 
135  Q_FOREACH (const DBlurEffectWidget *w, blurEffectWidgetList) {
136  if (!w->isVisible()) {
137  continue;
138  }
139 
140  QPainterPath p;
141  QRect r = w->rect();
142 
143  r.moveTopLeft(w->mapTo(topLevelWidget, r.topLeft()));
144  p.addRoundedRect(r, w->blurRectXRadius(), w->blurRectYRadius());
145 
146  if (!w->d_func()->maskPath.isEmpty()) {
147  p &= w->d_func()->maskPath.translated(r.topLeft());
148  }
149 
150  pathList << p;
151  }
152 
153  ok = DPlatformWindowHandle::setWindowBlurAreaByWM(topLevelWidget, pathList);
154  } else {
155  QVector<DPlatformWindowHandle::WMBlurArea> areaList;
156 
157  areaList.reserve(blurEffectWidgetList.size());
158 
159  Q_FOREACH (const DBlurEffectWidget *w, blurEffectWidgetList) {
160  if (!w->isVisible()) {
161  continue;
162  }
163 
164  QRect r = w->rect();
165 
166  r.moveTopLeft(w->mapTo(topLevelWidget, r.topLeft()));
167 
168  areaList << dMakeWMBlurArea(r.x(), r.y(), r.width(), r.height(), w->blurRectXRadius(), w->blurRectYRadius());
169  }
170 
171  ok = DPlatformWindowHandle::setWindowBlurAreaByWM(topLevelWidget, areaList);
172  }
173 
174  if (blurEffectWidgetList.isEmpty()) {
175  blurEffectWidgetHash.remove(topLevelWidget);
176  }
177 
178  return ok;
179 }

518 void DBlurEffectWidget::paintEvent(QPaintEvent *event)
519 {
520  D_D(DBlurEffectWidget);
521 
522  QPainter pa(this);
523 
524  if (d->blurRectXRadius > 0 || d->blurRectYRadius > 0) {
525  QPainterPath path;
526 
527  path.addRoundedRect(rect(), d->blurRectXRadius, d->blurRectYRadius);
528  pa.setRenderHint(QPainter::Antialiasing);
529  pa.setClipPath(path);
530  }
531 
532  if (!d->maskPath.isEmpty()) {
533  QPainterPath path = pa.clipPath();
534 
535  if (path.isEmpty()) {
536  path = d->maskPath;
537  } else {
538  path &= d->maskPath;
539  }
540 
541  pa.setClipPath(path);
542  }
543 
544  if (d->isBehindWindowBlendMode()) {
545  pa.setCompositionMode(QPainter::CompositionMode_Source);
546  } else {
547  int radius = d->radius;
548  QPoint point_offset = mapTo(window(), QPoint(0, 0));
549  const QRect &paintRect = event->rect();
550  qreal device_pixel_ratio = devicePixelRatioF();
551 
552  if (d->sourceImage.isNull()) {
553  const QRect &tmp_rect = rect().translated(point_offset).adjusted(-radius, -radius, radius, radius);
554 
555  d->sourceImage = window()->backingStore()->handle()->toImage().copy(tmp_rect * device_pixel_ratio);
556  d->sourceImage = d->sourceImage.scaledToWidth(d->sourceImage.width() / device_pixel_ratio);
557  } else {
558  QPainter pa_image(&d->sourceImage);
559 
560  pa_image.setCompositionMode(QPainter::CompositionMode_Source);
561 
562  for (const QRect &rect : event->region().rects()) {
563  if (device_pixel_ratio > 1) {
564  const QRect &tmp_rect = this->rect().translated(point_offset);
565  const QImage &area = window()->backingStore()->handle()->toImage().copy(tmp_rect * device_pixel_ratio);
566 
567  pa_image.drawImage(rect.topLeft() + QPoint(radius, radius), area.scaledToWidth(area.width() / device_pixel_ratio).copy(rect));
568  } else {
569  pa_image.drawImage(rect.topLeft() + QPoint(radius, radius),
570  window()->backingStore()->handle()->toImage().copy(rect.translated(point_offset)));
571  }
572  }
573 
574  pa_image.end();
575  }
576 
577  QImage image = d->sourceImage.copy(paintRect.adjusted(0, 0, 2 * radius, 2 * radius));
578 
579  QTransform old_transform = pa.transform();
580  pa.translate(paintRect.topLeft() - QPoint(radius, radius));
581  qt_blurImage(&pa, image, radius, false, false);
582  pa.setTransform(old_transform);
583  }
584 
585  pa.fillRect(rect(), maskColor());
586 }
