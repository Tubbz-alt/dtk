33 QImage dropShadow(const QPixmap &px, qreal radius, const QColor &color)
34 {
35  if (px.isNull()) {
36  return QImage();
37  }
38 
39  QSize size = px.size();
40 
41  QImage tmp(size + QSize(radius * 2, radius * 2), QImage::Format_ARGB32_Premultiplied);
42  tmp.fill(0);
43  QPainter tmpPainter(&tmp);
44  tmpPainter.setCompositionMode(QPainter::CompositionMode_Source);
45  tmpPainter.drawPixmap(QPoint(radius, radius), px);
46  tmpPainter.end();
47 
48  // blur the alpha channel
49  QImage blurred(tmp.size(), QImage::Format_ARGB32_Premultiplied);
50  blurred.fill(0);
51  QPainter blurPainter(&blurred);
52  qt_blurImage(&blurPainter, tmp, radius, false, true);
53  blurPainter.end();
54 
55  if (color == QColor(Qt::black)) {
56  return blurred;
57  }
58 
59  tmp = blurred;
60 
61  // blacken the image...
62  tmpPainter.begin(&tmp);
63  tmpPainter.setCompositionMode(QPainter::CompositionMode_SourceIn);
64  tmpPainter.fillRect(tmp.rect(), color);
65  tmpPainter.end();
66 
67  return tmp;
68 }
