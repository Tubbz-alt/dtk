116 static QWindow *ensureWindowHandle(QWidget *widget)
117 {
118  QWidget *window = widget->window();
119  QWindow *handle = window->windowHandle();
120 
121  if (!handle) {
122  bool save_flag = qApp->testAttribute(Qt::AA_DontCreateNativeWidgetSiblings);
123  qApp->setAttribute(Qt::AA_DontCreateNativeWidgetSiblings, true);
124  window->setAttribute(Qt::WA_NativeWindow);
125  handle = window->windowHandle();
126  window->setAttribute(Qt::WA_NativeWindow, false);
127 
128  // dxcb version >= 1.1.6

131  qApp->setAttribute(Qt::AA_DontCreateNativeWidgetSiblings, save_flag);
132  }
133  }

220 {
221  if (qFuzzyCompare(scale, 1.0))
222  return area;

224  DPlatformWindowHandle::WMBlurArea new_area;

226  new_area.x = qRound64(area.x * scale);

229  new_area.height = qRound64(area.height * scale);

236 bool DPlatformWindowHandle::setWindowBlurAreaByWM(QWindow *window, const QVector<DPlatformWindowHandle::WMBlurArea> &area)
237 {
238  if (!window) {
239  return false;
240  }
241 
242  if (isEnabledDXcb(window)) {
243  setWindowProperty(window, _windowBlurAreas, QVariant::fromValue(*(reinterpret_cast<const QVector<quint32>*>(&area))));
244 
245  return true;
246  }
247 
248  QFunctionPointer setWmBlurWindowBackgroundArea = Q_NULLPTR;
249 
250 #if QT_VERSION >= QT_VERSION_CHECK(5, 4, 0)
251  setWmBlurWindowBackgroundArea = qApp->platformFunction(_setWmBlurWindowBackgroundArea);
252 #endif
253 
254  if (!setWmBlurWindowBackgroundArea) {
255  qWarning("setWindowBlurAreaByWM is not support");
256 
257  return false;
258  }
259 
260  QSurfaceFormat format = window->format();
261 
262  format.setAlphaBufferSize(8);
263  window->setFormat(format);
264 
265  const qreal device_ratio = window->devicePixelRatio();
266 
267  if (qFuzzyCompare(device_ratio, 1.0)) {
268  return reinterpret_cast<bool(*)(const quint32, const QVector<WMBlurArea>&)>(setWmBlurWindowBackgroundArea)(window->winId(), area);
269  }
270 
271  QVector<WMBlurArea> new_areas;
272 
273  new_areas.reserve(area.size());
274 
275  for (const WMBlurArea &a : area) {
276  new_areas.append(a * device_ratio);
277  }
278 
279  return reinterpret_cast<bool(*)(const quint32, const QVector<WMBlurArea>&)>(setWmBlurWindowBackgroundArea)(window->winId(), new_areas);
280 }

282 inline QPainterPath operator *(const QPainterPath &path, qreal scale)
283 {
284  if (qFuzzyCompare(1.0, scale))
285  return path;
286 
287  QPainterPath new_path = path;
288 
289  for (int i = 0; i < path.elementCount(); ++i) {
290  const QPainterPath::Element &e = path.elementAt(i);
291 
292  new_path.setElementPositionAt(i, qRound(e.x * scale), qRound(e.y * scale));
293  }
294 
295  return new_path;

517 bool DPlatformWindowHandle::eventFilter(QObject *obj, QEvent *event)
518 {
519  if (obj == m_window) {
520  if (event->type() == QEvent::DynamicPropertyChange) {
521  QDynamicPropertyChangeEvent *e = static_cast<QDynamicPropertyChangeEvent *>(event);
522 
523  if (e->propertyName() == _windowRadius) {
524  Q_EMIT windowRadiusChanged();
525  } else if (e->propertyName() == _borderWidth) {
526  Q_EMIT borderWidthChanged();
527  } else if (e->propertyName() == _borderColor) {
528  Q_EMIT borderColorChanged();
529  } else if (e->propertyName() == _shadowRadius) {
530  Q_EMIT shadowRadiusChanged();
531  } else if (e->propertyName() == _shadowOffset) {
532  Q_EMIT shadowOffsetChanged();
533  } else if (e->propertyName() == _shadowColor) {
534  Q_EMIT shadowColorChanged();
535  } else if (e->propertyName() == _clipPath) {
536  Q_EMIT clipPathChanged();
537  } else if (e->propertyName() == _frameMask) {
538  Q_EMIT frameMaskChanged();
539  } else if (e->propertyName() == _frameMargins) {
540  Q_EMIT frameMarginsChanged();
541  } else if (e->propertyName() == _translucentBackground) {
542  Q_EMIT translucentBackgroundChanged();
543  } else if (e->propertyName() == _enableSystemResize) {
544  Q_EMIT enableSystemResizeChanged();
545  } else if (e->propertyName() == _enableSystemMove) {
546  Q_EMIT enableSystemMoveChanged();
547  } else if (e->propertyName() == _enableBlurWindow) {
548  Q_EMIT enableBlurWindowChanged();
549  } else if (e->propertyName() == _autoInputMaskByClipPath) {
550  Q_EMIT autoInputMaskByClipPathChanged();
551  }
552  }
553  }
554 
555  return false;
556 }
