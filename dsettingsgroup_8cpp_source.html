201 void DSettingsGroupPrivate::parseJson(const QString &prefixKey, const QJsonObject &group)
202 {
203  Q_Q(DSettingsGroup);
204  key = group.value("key").toString();
205  Q_ASSERT(!key.isEmpty());
206  key = prefixKey.isEmpty() ? key : prefixKey + "." + key;
207  name = group.value("name").toString();
208  hide = group.value("hide").toBool();
209 
210  for (auto optionJson : group.value("options").toArray()) {
211  auto optionObject = optionJson.toObject();
212  auto option = DSettingsOption::fromJson(key, optionObject);
213  options.insert(option->key(), option);
214  childOptions.insert(option->key(), option);
215  childOptionKeys << option->key();
216  }
217 
218  auto subGroups = group.value("groups").toArray();
219  for (auto subGroup : subGroups) {
220  auto child = DSettingsGroup::fromJson(key, subGroup.toObject());
221  child->setParentGroup(q);
222  childGroups.insert(child->key(), child);
223  childGroupKeys << child->key();
224 
225  for (auto option : child->options()) {
226  options.insert(option->key(), option);
227  }
228  }
229 
230 }
