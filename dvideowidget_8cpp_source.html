432 void DVideoWidget::paintEvent(QPaintEvent *)
433 {
434  D_DC(DVideoWidget);
435 
436  QPainter painter(this);
437 
438  const QMediaPlaylist *pl = d->player ? d->player->playlist() : NULL;
439  bool loop = pl && (pl->playbackMode() == QMediaPlaylist::Loop || pl->playbackMode() == QMediaPlaylist::CurrentItemInLoop);
440  QVideoFrame frame = (!loop || d->formatProxy->m_currentFrame.isValid()) ? d->formatProxy->m_currentFrame : d->formatProxy->m_lastFrame;
441 
442  frame.map(QAbstractVideoBuffer::ReadOnly);
443  QImage image(
444  frame.bits(),
445  frame.width(),
446  frame.height(),
447  frame.bytesPerLine(),
448  QVideoFrame::imageFormatFromPixelFormat(frame.pixelFormat()));
449  frame.unmap();
450 
451  if (image.isNull()) {
452  return;
453  }
454 
455  image.setDevicePixelRatio(d->ratio);
456  image = image.scaled(size() * d->scale * d->ratio, d->aspectRatioMode, Qt::SmoothTransformation);
457  image = image.mirrored(d->mirroredHorizontal, d->mirroredVertical);
458 
459  if (d->round) {
460  QPainterPath path;
461  int diameter = qMin(width(), height());
462  path.addEllipse(width()/2.0-diameter/2.0, height()/2.0-diameter/2.0, diameter, diameter);
463  painter.setRenderHint(QPainter::Antialiasing);
464  painter.setRenderHint(QPainter::SmoothPixmapTransform);
465  painter.setClipPath(path);
466  }
467 
468  painter.drawImage(QRectF(rect()).center() - QRectF(image.rect()).center() / d->ratio, image);
469 }
