249 void DSettingsOptionPrivate::parseJson(const QString &prefixKey, const QJsonObject &option)
250 {
251 // Q_Q(Option);
252  key = option.value("key").toString();
253  Q_ASSERT(!key.isEmpty());
254  Q_ASSERT(!prefixKey.isEmpty());
255  key = prefixKey + "." + key;
256  name = option.value("name").toString();
257 
258  canReset = !option.contains("reset") ? true : option.value("reset").toBool();
259  defalutValue = option.value("default").toVariant();
260  hidden = !option.contains("hide") ? false : option.value("hide").toBool();
261  viewType = option.value("type").toString();
262 
263  QStringList revserdKeys;
264  revserdKeys << "key" << "name" << "reset"
265  << "default" << "hide" << "type";
266 
267  auto allKeys = option.keys();
268  for (auto key : revserdKeys) {
269  allKeys.removeAll(key);
270  }
271 
272  for (auto key : allKeys) {
273  auto value = option.value(key);
274  if (value.isArray()) {
275  QStringList stringlist;
276  for (auto va : value.toArray()) {
277  stringlist << QString("%1").arg(va.toString());
278  }
279  datas.insert(key, stringlist);
280  } else {
281  datas.insert(key, value.toVariant());
282  }
283  }
284 }
