261 QList<DForeignWindow *> DWindowManagerHelper::currentWorkspaceWindows() const
262 {
263  D_DC(DWindowManagerHelper);
264 
265  for (QWindow *w : d->windowList) {
266  w->deleteLater();
267  }
268 
269  d->windowList.clear();
270 
271  QFunctionPointer wmClientList = Q_NULLPTR;
272 
273 #if QT_VERSION >= QT_VERSION_CHECK(5, 4, 0)
274  wmClientList = qApp->platformFunction(_getCurrentWorkspaceWindows);
275 #endif
276 
277  if (wmClientList) {
278  for (WId wid : reinterpret_cast<QVector<quint32>(*)()>(wmClientList)()) {
279  if (DForeignWindow *w = DForeignWindow::fromWinId(wid)) {
280  d->windowList << w;
281  }
282  }
283  }
284 
285  return d->windowList;
286 }
