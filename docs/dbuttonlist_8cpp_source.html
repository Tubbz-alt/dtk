164 void DButtonList::addButton(const QString &label, int index){
165  DIconButton* button = new DIconButton(":/images/dark/images/tick_hover.png", label, this);
166  button->setSizePolicy(QSizePolicy::Expanding, QSizePolicy::Expanding);
167  button->setCheckable(true);
168  connect(button, SIGNAL(mouseEntered(QString)), this, SIGNAL(buttonMouseEntered(QString)));
169  connect(button, SIGNAL(mouseLeaved(QString)), this, SIGNAL(buttonMouseLeaved(QString)));
170  QFrame* borderFrame = new QFrame;
171  borderFrame->setObjectName("BorderFrame");
172  QVBoxLayout* borderLayout = new QVBoxLayout;
173  borderLayout->addWidget(button);
174  borderLayout->setSpacing(0);
175  borderLayout->setContentsMargins(m_leftMargin, 0, m_rightMargin, 0);
176  borderFrame->setLayout(borderLayout);
177 
178  m_buttonGroup->addButton(button, index);
179  QListWidgetItem* item = new QListWidgetItem(this);
180  addItem(item);
181  setItemWidget(item, borderFrame);
182 
183  setItemSize(gridSize());
184 
185 }

194 void DButtonList::setButtonChecked(int id){
195  if (id < m_buttonGroup->buttons().length()){
196  DIconButton* button = reinterpret_cast<DIconButton*>(m_buttonGroup->button(id));
197  button->setChecked(true);
198  if (m_buttonGroup->buttons().length() == 1){
199  button->setProperty("state", "OnlyOne");
200  button->hideIconLabel();
201  }else{
202  button->setProperty("state", "normal");
203  }
204  button->updateStyle();
205  Q_EMIT buttonCheckedIndexChanged(id);
206  Q_EMIT buttonChecked(button->text());
207  }
208 }
