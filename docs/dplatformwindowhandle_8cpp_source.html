797 bool DPlatformWindowHandle::setWindowBlurAreaByWM(QWindow *window, const QVector<DPlatformWindowHandle::WMBlurArea> &area)
798 {
799  if (!window) {
800  return false;
801  }
802 
803  if (isEnabledDXcb(window)) {
804  setWindowProperty(window, _windowBlurAreas, QVariant::fromValue(*(reinterpret_cast<const QVector<quint32>*>(&area))));
805 
806  return true;
807  }
808 
809  QFunctionPointer setWmBlurWindowBackgroundArea = Q_NULLPTR;
810 
811 #if QT_VERSION >= QT_VERSION_CHECK(5, 4, 0)
812  setWmBlurWindowBackgroundArea = qApp->platformFunction(_setWmBlurWindowBackgroundArea);
813 #endif
814 
815  if (!setWmBlurWindowBackgroundArea) {
816  qWarning("setWindowBlurAreaByWM is not support");
817 
818  return false;
819  }
820 
821  QSurfaceFormat format = window->format();
822 
823  format.setAlphaBufferSize(8);
824  window->setFormat(format);
825 
826  const qreal device_ratio = window->devicePixelRatio();
827 
828  if (qFuzzyCompare(device_ratio, 1.0)) {
829  return reinterpret_cast<bool(*)(const quint32, const QVector<WMBlurArea>&)>(setWmBlurWindowBackgroundArea)(window->winId(), area);
830  }
831 
832  QVector<WMBlurArea> new_areas;
833 
834  new_areas.reserve(area.size());
835 
836  for (const WMBlurArea &a : area) {
837  new_areas.append(a * device_ratio);
838  }
839 
840  return reinterpret_cast<bool(*)(const quint32, const QVector<WMBlurArea>&)>(setWmBlurWindowBackgroundArea)(window->winId(), new_areas);
841 }

843 inline QPainterPath operator *(const QPainterPath &path, qreal scale)
844 {
845  if (qFuzzyCompare(1.0, scale))
846  return path;
847 
848  QPainterPath new_path = path;
849 
850  for (int i = 0; i < path.elementCount(); ++i) {
851  const QPainterPath::Element &e = path.elementAt(i);
852 
853  new_path.setElementPositionAt(i, qRound(e.x * scale), qRound(e.y * scale));
854  }
855 
856  return new_path;
857 }
