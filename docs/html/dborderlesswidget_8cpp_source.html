66 void DBorderlessWidgetPrivate::init()
67 {
68  D_Q(DBorderlessWidget);
69 
70  dwindowFlags = Qt::Window | Qt::WindowTitleHint |
71  Qt::WindowSystemMenuHint | Qt::WindowMinMaxButtonsHint |
72  Qt::WindowCloseButtonHint | Qt::WindowFullscreenButtonHint;
73  m_NormalShadowWidth = WindowGlowRadius;
74  m_ShadowWidth = WindowGlowRadius;
75  shadowColor = SHADOW_COLOR_ACTIVE;
76  shadowOffset = QPoint(0, 10);
77  m_Radius = WindowRadius;
78  m_Border = WindowBorder;
79  m_ResizeHandleWidth = WindowHandleWidth;
80  m_MousePressed = false;
81  m_backgroundColor = BackgroundTopColor;
82 
83  rootLayout = new QVBoxLayout(q);
84 
85  rootLayout->setSpacing(0);
86  updateContentsMargins();
87 
88  windowWidget = new QWidget;
89  QVBoxLayout *windowLayout = new QVBoxLayout(windowWidget);
90  windowLayout->setMargin(0);
91  windowLayout->setSpacing(0);
92 
93  rootLayout->addWidget(windowWidget);
94 
95  titlebar = new DTitlebar;
96  contentWidget = new QWidget;
97  QVBoxLayout *contentWidgetLayout = new QVBoxLayout;
98  contentWidgetLayout->setSpacing(0);
99  contentWidgetLayout->setMargin(0);
100  contentWidget->setLayout(contentWidgetLayout);
101  contentWidget->setContentsMargins(m_Border, 0, m_Border, m_Border);
102  contentWidget->setSizePolicy(QSizePolicy::Expanding, QSizePolicy::Expanding);
103 
104  windowLayout->addWidget(titlebar);
105  windowLayout->addWidget(contentWidget);
106 
107  auto filter = new FilterMouseMove(windowWidget);
108  windowWidget->installEventFilter(filter);
109  filter->m_rootWidget = q;
110 
111  q->connect(titlebar, &DTitlebar::optionClicked, q, &DBorderlessWidget::optionClicked);
112  q->connect(titlebar, &DTitlebar::mouseMoving, q, &DBorderlessWidget::moveWindow);
113 
114  q->connect(titlebar, SIGNAL(mousePressed(Qt::MouseButtons)), q, SLOT(_q_onTitleBarMousePressed(Qt::MouseButtons)));
115 
116  q->connect(qApp, &QGuiApplication::focusWindowChanged, q, [q] {
117  if (q->isActiveWindow())
118  {
119  q->setShadowColor(SHADOW_COLOR_ACTIVE);
120  } else
121  {
122  q->setShadowColor(SHADOW_COLOR_NORMAL);
123  }
124  });
125 }

836 void DBorderlessWidget::setBackgroundImage(const QPixmap &srcPixmap)
837 {
838  D_D(DBorderlessWidget);
839 
840  int radius = d->m_Radius;
841  int windowExtern = d->m_ShadowWidth + d->m_Border * 2;
842  QRect windowRect = QWidget::rect().marginsRemoved(
843  QMargins(windowExtern, windowExtern, windowExtern, windowExtern));
844  QSize sz = windowRect.size();
845  QPixmap backgroundPixmap = srcPixmap.scaled(sz, Qt::KeepAspectRatioByExpanding);
846 
847  QPixmap maskPixmap(sz);
848  maskPixmap.fill(Qt::transparent);
849  QPainterPath path;
850  path.addRoundRect(QRectF(0, 0, sz.width(), sz.height()), radius / 2);
851  QPainter bkPainter(&maskPixmap);
852  bkPainter.setRenderHint(QPainter::Antialiasing);
853  bkPainter.setPen(QPen(Qt::white, 1));
854  bkPainter.fillPath(path, QBrush(Qt::red));
855 
856  QPainter::CompositionMode mode = QPainter::CompositionMode_SourceIn;
857  QImage resultImage = QImage(sz, QImage::Format_ARGB32_Premultiplied);
858  QPainter painter(&resultImage);
859  painter.setCompositionMode(QPainter::CompositionMode_Source);
860  painter.fillRect(resultImage.rect(), Qt::transparent);
861  painter.setCompositionMode(QPainter::CompositionMode_SourceOver);
862  painter.drawImage(0, 0, maskPixmap.toImage());
863  painter.setCompositionMode(mode);
864  painter.drawImage(0, 0, backgroundPixmap.toImage());
865  painter.setCompositionMode(QPainter::CompositionMode_DestinationOver);
866  painter.end();
867 
868  d->m_Background = QPixmap::fromImage(resultImage);
869 }
