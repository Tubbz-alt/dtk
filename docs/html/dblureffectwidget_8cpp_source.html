124 bool DBlurEffectWidgetPrivate::updateWindowBlurArea(QWidget *topLevelWidget)
125 {
126  if (!topLevelWidget->isVisible()) {
127  return false;
128  }
129 
130  QList<const DBlurEffectWidget *> blurEffectWidgetList = blurEffectWidgetHash.values(topLevelWidget);
131 
132  bool isExistMaskPath = false;
133 
134  Q_FOREACH (const DBlurEffectWidget *w, blurEffectWidgetList) {
135  if (!w->d_func()->blurEnabled) {
136  continue;
137  }
138 
139  // 当存在一个模糊区域填充整个窗口时，不需要再考虑其它的控件，直接对此窗口开启全模糊即可
140  if (w->d_func()->isFull() && w->isVisible()) {
141  DPlatformWindowHandle handle(topLevelWidget);
142 
143  if (!handle.enableBlurWindow()) {
144  handle.setEnableBlurWindow(true);
145  }
146 
147  return true;
148  }
149 
150  if (!w->d_func()->maskPath.isEmpty() && w->isVisible()) {
151  isExistMaskPath = true;
152  break;
153  }
154  }
155 
156  // 清理窗口背景模糊是否完全开启的状态
157  DPlatformWindowHandle handle(topLevelWidget);
158 
159  if (handle.enableBlurWindow()) {
160  handle.setEnableBlurWindow(false);
161  }
162 
163  bool ok = false;
164 
165  if (isExistMaskPath) {
166  QList<QPainterPath> pathList;
167 
168  Q_FOREACH (const DBlurEffectWidget *w, blurEffectWidgetList) {
169  if (!w->d_func()->blurEnabled) {
170  continue;
171  }
172 
173  if (!w->isVisible()) {
174  continue;
175  }
176 
177  QPainterPath p;
178  QRect r = w->rect();
179 
180  r.moveTopLeft(w->mapTo(topLevelWidget, r.topLeft()));
181  p.addRoundedRect(r, w->blurRectXRadius(), w->blurRectYRadius());
182 
183  if (!w->d_func()->maskPath.isEmpty()) {
184  p &= w->d_func()->maskPath.translated(r.topLeft());
185  }
186 
187  pathList << p;
188  }
189 
190  ok = handle.setWindowBlurAreaByWM(pathList);
191  } else {
192  QVector<DPlatformWindowHandle::WMBlurArea> areaList;
193 
194  areaList.reserve(blurEffectWidgetList.size());
195 
196  Q_FOREACH (const DBlurEffectWidget *w, blurEffectWidgetList) {
197  if (!w->d_func()->blurEnabled) {
198  continue;
199  }
200 
201  if (!w->isVisible()) {
202  continue;
203  }
204 
205  QRect r = w->rect();
206 
207  r.moveTopLeft(w->mapTo(topLevelWidget, r.topLeft()));
208 
209  areaList << dMakeWMBlurArea(r.x(), r.y(), r.width(), r.height(), w->blurRectXRadius(), w->blurRectYRadius());
210  }
211 
212  ok = handle.setWindowBlurAreaByWM(areaList);
213  }
214 
215  if (blurEffectWidgetList.isEmpty()) {
216  blurEffectWidgetHash.remove(topLevelWidget);
217  }
218 
219  return ok;
220 }

763 void DBlurEffectWidget::paintEvent(QPaintEvent *event)
764 {
765  D_D(DBlurEffectWidget);
766 
767  if (!d->blurEnabled)
768  return;
769 
770  QPainter pa(this);
771 
772  if (d->blurRectXRadius > 0 || d->blurRectYRadius > 0) {
773  QPainterPath path;
774 
775  path.addRoundedRect(rect(), d->blurRectXRadius, d->blurRectYRadius);
776  pa.setRenderHint(QPainter::Antialiasing);
777  pa.setClipPath(path);
778  }
779 
780  if (!d->maskPath.isEmpty()) {
781  QPainterPath path = pa.clipPath();
782 
783  if (path.isEmpty()) {
784  path = d->maskPath;
785  } else {
786  path &= d->maskPath;
787  }
788 
789  pa.setClipPath(path);
790  }
791 
792  if (d->isBehindWindowBlendMode()) {
793  pa.setCompositionMode(QPainter::CompositionMode_Source);
794  } else {
795  int radius = d->radius;
796  QPoint point_offset = mapTo(window(), QPoint(0, 0));
797  const QRect &paintRect = event->rect();
798  qreal device_pixel_ratio = devicePixelRatioF();
799 
800  if (d->sourceImage.isNull()) {
801  const QRect &tmp_rect = rect().translated(point_offset).adjusted(-radius, -radius, radius, radius);
802 
803  d->sourceImage = window()->backingStore()->handle()->toImage().copy(tmp_rect * device_pixel_ratio);
804  d->sourceImage = d->sourceImage.scaledToWidth(d->sourceImage.width() / device_pixel_ratio);
805  } else {
806  QPainter pa_image(&d->sourceImage);
807 
808  pa_image.setCompositionMode(QPainter::CompositionMode_Source);
809 
810  for (const QRect &rect : event->region().rects()) {
811  if (device_pixel_ratio > 1) {
812  const QRect &tmp_rect = this->rect().translated(point_offset);
813  const QImage &area = window()->backingStore()->handle()->toImage().copy(tmp_rect * device_pixel_ratio);
814 
815  pa_image.drawImage(rect.topLeft() + QPoint(radius, radius), area.scaledToWidth(area.width() / device_pixel_ratio).copy(rect));
816  } else {
817  pa_image.drawImage(rect.topLeft() + QPoint(radius, radius),
818  window()->backingStore()->handle()->toImage().copy(rect.translated(point_offset)));
819  }
820  }
821 
822  pa_image.end();
823  }
824 
825  QImage image = d->sourceImage.copy(paintRect.adjusted(0, 0, 2 * radius, 2 * radius));
826 
827  QTransform old_transform = pa.transform();
828  pa.translate(paintRect.topLeft() - QPoint(radius, radius));
829  qt_blurImage(&pa, image, radius, false, false);
830  pa.setTransform(old_transform);
831  }
832 
833  pa.fillRect(rect(), maskColor());
834 }
