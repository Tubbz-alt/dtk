473 {
474  // 优先使用窗口管理器中实现的no titlebar接口实现自定义窗口修饰器的效果

492  "Must be called before window handle has been created. See also QWindow::handle()");
493  } else {
494  window->setProperty(_useDxcb, true);

657 bool DPlatformHandle::setWindowBlurAreaByWM(QWindow *window, const QVector<DPlatformHandle::WMBlurArea> &area)
658 {
659  if (!window) {
660  return false;
661  }
662 
663  if (isEnabledDXcb(window)) {
664  setWindowProperty(window, _windowBlurAreas, QVariant::fromValue(*(reinterpret_cast<const QVector<quint32>*>(&area))));
665 
666  return true;
667  }
668 
669  QFunctionPointer setWmBlurWindowBackgroundArea = Q_NULLPTR;
670 
671 #if QT_VERSION >= QT_VERSION_CHECK(5, 4, 0)
672  setWmBlurWindowBackgroundArea = qApp->platformFunction(_setWmBlurWindowBackgroundArea);
673 #endif
674 
675  if (!setWmBlurWindowBackgroundArea) {
676  qWarning("setWindowBlurAreaByWM is not support");
677 
678  return false;
679  }
680 
681  QSurfaceFormat format = window->format();
682 
683  format.setAlphaBufferSize(8);
684  window->setFormat(format);
685 
686  const qreal device_ratio = window->devicePixelRatio();
687 
688  if (qFuzzyCompare(device_ratio, 1.0)) {
689  return reinterpret_cast<bool(*)(const quint32, const QVector<WMBlurArea>&)>(setWmBlurWindowBackgroundArea)(window->winId(), area);
690  }
691 
692  QVector<WMBlurArea> new_areas;
693 
694  new_areas.reserve(area.size());
695 
696  for (const WMBlurArea &a : area) {
697  new_areas.append(a * device_ratio);
698  }
699 
700  return reinterpret_cast<bool(*)(const quint32, const QVector<WMBlurArea>&)>(setWmBlurWindowBackgroundArea)(window->winId(), new_areas);
701 }

703 inline QPainterPath operator *(const QPainterPath &path, qreal scale)
704 {
705  if (qFuzzyCompare(1.0, scale))
706  return path;
707 
708  QPainterPath new_path = path;
709 
710  for (int i = 0; i < path.elementCount(); ++i) {
711  const QPainterPath::Element &e = path.elementAt(i);
712 
713  new_path.setElementPositionAt(i, qRound(e.x * scale), qRound(e.y * scale));
714  }
715 
716  return new_path;
717 }

1037 bool DPlatformHandle::eventFilter(QObject *obj, QEvent *event)
1038 {
1039  if (obj == m_window) {
1040  if (event->type() == QEvent::DynamicPropertyChange) {
1041  QDynamicPropertyChangeEvent *e = static_cast<QDynamicPropertyChangeEvent *>(event);
1042 
1043  if (e->propertyName() == _windowRadius) {
1044  Q_EMIT windowRadiusChanged();
1045  } else if (e->propertyName() == _borderWidth) {
1046  Q_EMIT borderWidthChanged();
1047  } else if (e->propertyName() == _borderColor) {
1048  Q_EMIT borderColorChanged();
1049  } else if (e->propertyName() == _shadowRadius) {
1050  Q_EMIT shadowRadiusChanged();
1051  } else if (e->propertyName() == _shadowOffset) {
1052  Q_EMIT shadowOffsetChanged();
1053  } else if (e->propertyName() == _shadowColor) {
1054  Q_EMIT shadowColorChanged();
1055  } else if (e->propertyName() == _clipPath) {
1056  Q_EMIT clipPathChanged();
1057  } else if (e->propertyName() == _frameMask) {
1058  Q_EMIT frameMaskChanged();
1059  } else if (e->propertyName() == _frameMargins) {
1060  Q_EMIT frameMarginsChanged();
1061  } else if (e->propertyName() == _translucentBackground) {
1062  Q_EMIT translucentBackgroundChanged();
1063  } else if (e->propertyName() == _enableSystemResize) {
1064  Q_EMIT enableSystemResizeChanged();
1065  } else if (e->propertyName() == _enableSystemMove) {
1066  Q_EMIT enableSystemMoveChanged();
1067  } else if (e->propertyName() == _enableBlurWindow) {
1068  Q_EMIT enableBlurWindowChanged();
1069  } else if (e->propertyName() == _autoInputMaskByClipPath) {
1070  Q_EMIT autoInputMaskByClipPathChanged();
1071  }
1072  }
1073  }
1074 
1075  return false;
1076 }
