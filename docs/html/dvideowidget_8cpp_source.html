684 void DVideoWidget::paintEvent(QPaintEvent *)
685 {
686  D_DC(DVideoWidget);
687 
688  QPainter painter(this);
689 
690  const QMediaPlaylist *pl = d->player ? d->player->playlist() : NULL;
691  bool loop = pl && (pl->playbackMode() == QMediaPlaylist::Loop || pl->playbackMode() == QMediaPlaylist::CurrentItemInLoop);
692  QVideoFrame frame = (!loop || d->formatProxy->m_currentFrame.isValid()) ? d->formatProxy->m_currentFrame : d->formatProxy->m_lastFrame;
693 
694  frame.map(QAbstractVideoBuffer::ReadOnly);
695  QImage image(
696  frame.bits(),
697  frame.width(),
698  frame.height(),
699  frame.bytesPerLine(),
700  QVideoFrame::imageFormatFromPixelFormat(frame.pixelFormat()));
701  frame.unmap();
702 
703  if (image.isNull()) {
704  return;
705  }
706 
707  image.setDevicePixelRatio(d->ratio);
708  image = image.scaled(size() * d->scale * d->ratio, d->aspectRatioMode, Qt::SmoothTransformation);
709  image = image.mirrored(d->mirroredHorizontal, d->mirroredVertical);
710 
711  if (d->round) {
712  QPainterPath path;
713  int diameter = qMin(width(), height());
714  path.addEllipse(width()/2.0-diameter/2.0, height()/2.0-diameter/2.0, diameter, diameter);
715  painter.setRenderHint(QPainter::Antialiasing);
716  painter.setRenderHint(QPainter::SmoothPixmapTransform);
717  painter.setClipPath(path);
718  }
719 
720  painter.drawImage(QRectF(rect()).center() - QRectF(image.rect()).center() / d->ratio, image);
721 }
