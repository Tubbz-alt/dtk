253 void DTitlebarPrivate::updateFullscreen()
254 {
255  D_Q(DTitlebar);
256 
257  if (!autoHideOnFullscreen) {
258  return;
259  }
260 
261  bool isFullscreen = targetWindow()->windowState().testFlag(Qt::WindowFullScreen);
262  auto mainWindow = qobject_cast<DMainWindow *>(targetWindow());
263  if (!isFullscreen) {
264  quitFullButton->hide();
265  mainWindow->setMenuWidget(q);
266  showOnFullscreen();
267  } else {
268  // must set to empty
269  quitFullButton->show();
270  if (mainWindow->menuWidget()) {
271  mainWindow->menuWidget()->setParent(nullptr);
272  mainWindow->setMenuWidget(Q_NULLPTR);
273  }
274 
275  q->setParent(mainWindow);
276  q->show();
277  hideOnFullscreen();
278  }

281 void DTitlebarPrivate::updateButtonsState(Qt::WindowFlags type)
282 {
283  D_Q(DTitlebar);
284  bool useDXcb = DPlatformWindowHandle::isEnabledDXcb(targetWindow());
285  bool isFullscreen = targetWindow()->windowState().testFlag(Qt::WindowFullScreen);
286 
287  bool forceShow = !useDXcb;
288 #ifndef Q_OS_LINUX
289  forceShow = false;
290 #endif
291 
292  bool showTitle = (type.testFlag(Qt::WindowTitleHint) || forceShow) && !embedMode;
293  if (titleLabel) {
294  titleLabel->setVisible(showTitle);
295  }
296  if (iconLabel) {
297  iconLabel->setVisible(showTitle);
298  }
299 
300  // Never show in embed/fullscreen
301  bool forceHide = embedMode || isFullscreen;
302 
303  bool showMin = (type.testFlag(Qt::WindowMinimizeButtonHint) || forceShow) && !forceHide;
304  minButton->setVisible(showMin);
305 
306 
307  bool allowResize = true;
308 
309  if (q->window() && q->window()->windowHandle()) {
310  auto functions_hints = DWindowManagerHelper::getMotifFunctions(q->window()->windowHandle());
311  allowResize = functions_hints.testFlag(DWindowManagerHelper::FUNC_RESIZE);
312  }
313 
314  bool showMax = (type.testFlag(Qt::WindowMaximizeButtonHint) || forceShow) && !forceHide && allowResize;
315 // qDebug() << "max:"
316 // << "allowResize" << allowResize
317 // << "useDXcb" << useDXcb
318 // << "forceHide" << forceHide
319 // << "type.testFlag(Qt::WindowMaximizeButtonHint)" << type.testFlag(Qt::WindowMaximizeButtonHint);
320  maxButton->setVisible(showMax);
321 
322  bool showClose = (type.testFlag(Qt::WindowCloseButtonHint) || forceShow) && !forceHide;
323  closeButton->setVisible(showClose);
324 
325  buttonArea->adjustSize();
326  buttonArea->resize(buttonArea->size());
327 
328  if (titlePadding) {
329  titlePadding->setFixedSize(buttonArea->size());
330  }
331 }

333 void DTitlebarPrivate::updateButtonsFunc()
334 {
335  if (!targetWindowHandle) {
336  return;
337  }
338 
339  // 根据 disableFlags 更新窗口标志，而标题栏上具体按钮的开启/禁用状态只根据
340  // 窗口标志改变后做更新，且窗口标志改变后也会同步更新 disableFlags 的值，
341  // 也就是说，实际上窗口按钮的状态只受窗口标志影响，而 setDisableFlags
342  // 只是单纯的记录新的 disableFlags 的值，由于调用 setDisableFlags
343  // 是应用中的主动行为，所以同时也会根据这个值更新窗口标志。
344  DWindowManagerHelper::setMotifFunctions(
345  targetWindowHandle,
346  DWindowManagerHelper::FUNC_MAXIMIZE,
347  !disableFlags.testFlag(Qt::WindowMaximizeButtonHint));
348  DWindowManagerHelper::setMotifFunctions(
349  targetWindowHandle,
350  DWindowManagerHelper::FUNC_MINIMIZE,
351  !disableFlags.testFlag(Qt::WindowMinimizeButtonHint));
352  DWindowManagerHelper::setMotifFunctions(
353  targetWindowHandle,
354  DWindowManagerHelper::FUNC_CLOSE,
355  !disableFlags.testFlag(Qt::WindowCloseButtonHint));
356 }

411 void DTitlebarPrivate::_q_onTopWindowMotifHintsChanged(quint32 winId)
412 {
413  D_QC(DTitlebar);
414 
415  if (!DPlatformWindowHandle::isEnabledDXcb(targetWindow())) {
416  q->disconnect(DWindowManagerHelper::instance(), SIGNAL(windowMotifWMHintsChanged(quint32)),
417  q, SLOT(_q_onTopWindowMotifHintsChanged(quint32)));
418  return;
419  }
420 
421  if (winId != q->window()->internalWinId()) {
422  return;
423  }
424 
425  DWindowManagerHelper::MotifDecorations decorations_hints = DWindowManagerHelper::getMotifDecorations(q->window()->windowHandle());
426  DWindowManagerHelper::MotifFunctions functions_hints = DWindowManagerHelper::getMotifFunctions(q->window()->windowHandle());
427 
428  if (titleLabel) {
429  titleLabel->setVisible(decorations_hints.testFlag(DWindowManagerHelper::DECOR_TITLE));
430  }
431 
432  if (iconLabel) {
433  iconLabel->setVisible(decorations_hints.testFlag(DWindowManagerHelper::DECOR_TITLE));
434  }
435 
436  updateButtonsState(targetWindow()->windowFlags());
437 
438  minButton->setEnabled(functions_hints.testFlag(DWindowManagerHelper::FUNC_MINIMIZE));
439  maxButton->setEnabled(functions_hints.testFlag(DWindowManagerHelper::FUNC_MAXIMIZE)
440  && functions_hints.testFlag(DWindowManagerHelper::FUNC_RESIZE));
441  closeButton->setEnabled(functions_hints.testFlag(DWindowManagerHelper::FUNC_CLOSE));
442  // sync button state
443 #if QT_VERSION >= QT_VERSION_CHECK(5, 7, 0)
444  disableFlags.setFlag(Qt::WindowMinimizeButtonHint, !minButton->isEnabled());
445  disableFlags.setFlag(Qt::WindowMaximizeButtonHint, !maxButton->isEnabled());
446  disableFlags.setFlag(Qt::WindowCloseButtonHint, !closeButton->isEnabled());
447 #else
448  setWindowFlag(disableFlags, Qt::WindowMinimizeButtonHint, !minButton->isEnabled());
449  setWindowFlag(disableFlags, Qt::WindowMaximizeButtonHint, !maxButton->isEnabled());
450  setWindowFlag(disableFlags, Qt::WindowCloseButtonHint, !closeButton->isEnabled());
451 #endif
452 
453  if (titlePadding) {
454  titlePadding->setFixedSize(buttonArea->size());
455  }
456 }

690 bool DTitlebar::eventFilter(QObject *obj, QEvent *event)
691 {
692  D_D(DTitlebar);
693 
694  if (obj == d->targetWindow()) {
695  switch (event->type()) {
696  case QEvent::ShowToParent:
697  d->handleParentWindowIdChange();
698  d->updateButtonsState(d->targetWindow()->windowFlags());
699  break;
700  case QEvent::Resize:
701  if (d->autoHideOnFullscreen) {
702  setFixedWidth(d->targetWindow()->width());
703  }
704  break;
705  case QEvent::HoverMove: {
706  auto mouseEvent = reinterpret_cast<QMouseEvent *>(event);
707  bool isFullscreen = d->targetWindow()->windowState().testFlag(Qt::WindowFullScreen);
708  if (isFullscreen && d->autoHideOnFullscreen) {
709  if (mouseEvent->pos().y() > height() && d->isVisableOnFullscreen()) {
710  d->hideOnFullscreen();
711  }
712  if (mouseEvent->pos().y() < 2) {
713  d->showOnFullscreen();
714  }
715  }
716  break;
717  }
718  case QEvent::WindowStateChange: {
719  d->handleParentWindowStateChange();
720  break;
721  }
722  default:
723  break;
724  }
725  }
726  return QWidget::eventFilter(obj, event);
727 }
