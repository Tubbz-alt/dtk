122 static bool tryAcquireSystemSemaphore(QSystemSemaphore *ss, qint64 timeout = 10)
123 {
124  if (ss->error() != QSystemSemaphore::NoError) {
125  return false;
126  }
127 
128  QSystemSemaphore _tmp_ss(QString("%1-%2").arg("DTK::tryAcquireSystemSemaphore").arg(ss->key()), 1, QSystemSemaphore::Open);
129 
130  _tmp_ss.acquire();
131 
132  QElapsedTimer t;
133  QFuture<bool> request = QtConcurrent::run(ss, &QSystemSemaphore::acquire);
134 
135  t.start();
136 
137  while (Q_LIKELY(t.elapsed() < timeout && !request.isFinished()));
138 
139  if (request.isFinished()) {
140  return true;
141  }
142 
143  if (Q_LIKELY(request.isRunning())) {
144  if (Q_LIKELY(ss->release(1))) {
145  request.waitForFinished();
146  }
147  }
148 
149  return false;
150 }

152 bool DApplicationPrivate::setSingleInstanceBySemaphore(const QString &key)
153 {
154  static QSystemSemaphore ss(key, 1, QSystemSemaphore::Open);
155  static bool singleInstance = false;
156 
157  if (singleInstance) {
158  return true;
159  }
160 
161  Q_ASSERT_X(ss.error() == QSystemSemaphore::NoError, "DApplicationPrivate::setSingleInstanceBySemaphore:", ss.errorString().toLocal8Bit().constData());
162 
163  singleInstance = tryAcquireSystemSemaphore(&ss);
164 
165  if (singleInstance) {
166  QtConcurrent::run([this] {
167  QPointer<DApplication> that = q_func();
168 
169  while (ss.acquire() && singleInstance)
170  {
171  if (!that) {
172  return;
173  }
174 
175  if (that->startingUp() || that->closingDown()) {
176  break;
177  }
178 
179  ss.release(1);
180 
181  if (that) {
182  Q_EMIT that->newInstanceStarted();
183  }
184  }
185  });
186 
187  auto clean_fun = [] {
188  ss.release(1);
189  singleInstance = false;
190  };
191 
192  qAddPostRoutine(clean_fun);
193  atexit(clean_fun);
194  }
195 
196  return singleInstance;
197 }

255 bool DApplicationPrivate::loadTranslator(QList<DPathBuf> translateDirs, const QString &name, QList<QLocale> localeFallback)
256 {
257  D_Q(DApplication);
258 
259  for (auto &locale : localeFallback) {
260  QString translateFilename = QString("%1_%2").arg(name).arg(locale.name());
261  for (auto &path : translateDirs) {
262  QString translatePath = (path / translateFilename).toString();
263  if (QFile::exists(translatePath + ".qm")) {
264  qDebug() << "load translate" << translatePath;
265  auto translator = new QTranslator(q);
266  translator->load(translatePath);
267  q->installTranslator(translator);
268  return true;
269  }
270  }
271 
272  QStringList parseLocalNameList = locale.name().split("_", QString::SkipEmptyParts);
273  if (parseLocalNameList.length() > 0) {
274  translateFilename = QString("%1_%2").arg(name)
275  .arg(parseLocalNameList.at(0));
276  for (auto &path : translateDirs) {
277  QString translatePath = (path / translateFilename).toString();
278  if (QFile::exists(translatePath + ".qm")) {
279  qDebug() << "translatePath after feedback:" << translatePath;
280  auto translator = new QTranslator(q);
281  translator->load(translatePath);
282  q->installTranslator(translator);
283  return true;
284  }
285  }
286  }
287  }
288 
289  qWarning() << name << "can not find qm files" ;
290  return false;
291 }

403  // set QTapAndHoldGesture::timeout
404  if (QGSettings::isSchemaInstalled("com.deepin.dde.touchscreen")) {
405  QGSettings gsettings("com.deepin.dde.touchscreen");
406  if (gsettings.keys().contains("longpressDuration"))
407  // NOTE(hualet): -100 is a workaround against the situation that that sometimes longpress
408  // and release on Dock cause App launches which should be avoided.
409  //
410  // I guess it happens like this: longpress happens on Dock,
411  // Dock menu shows(doesn't grab the mouse which is a bug can't be fixed easily),
412  // user ends the longpress, DDE Dock recevies mouseReleaseEvent and checks for
413  // QTapAndHoldGesture which is still not happening (maybe because the timer used is
414  // a CoarseTimer?), so Dock treats the event as a normal mouseReleaseEvent, launches the
415  // App or triggers the action.
416  //
417  // see: https://github.com/linuxdeepin/internal-discussion/issues/430
418  //
419  // This workaround hopefully can fix most of this situations.
420  QTapAndHoldGesture::setTimeout(gsettings.get("longpress-duration").toInt() - 100);
421  }
422 #endif
