180 void DFileChooserEditPrivate::_q_showFileChooserDialog()
181 {
182  D_Q(DFileChooserEdit);
183 
184  QFileDialog dialog(q);
185 
186  dialog.setAcceptMode(QFileDialog::AcceptOpen);
187  dialog.setFileMode(fileMode);
188  dialog.setNameFilters(nameFilters);
189 
190  if (dialogDisplayPosition == DFileChooserEdit::CurrentMonitorCenter) {
191  QPoint pos = QCursor::pos();
192 
193  for (QScreen *screen : qApp->screens()) {
194  if (screen->geometry().contains(pos)) {
195  QRect rect = dialog.geometry();
196  rect.moveCenter(screen->geometry().center());
197  dialog.move(rect.topLeft());
198  break;
199  }
200  }
201  }
202 
203  q->dialogOpened();
204 
205  int code = dialog.exec();
206 
207 
208  if (code == QDialog::Accepted && !dialog.selectedFiles().isEmpty()) {
209  const QString fileName = dialog.selectedFiles().first();
210 
211  q->setText(fileName);
212  Q_EMIT q->fileChoosed(fileName);
213  }
214 
215  q->dialogClosed(code);
216 }
