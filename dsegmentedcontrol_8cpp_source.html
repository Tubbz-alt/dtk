331 bool DSegmentedControl::setCurrentIndex(int currentIndex)
332 {
333  if(currentIndex == m_currentIndex)
334  return true;
335 
336  if(currentIndex<0||currentIndex>count()-1){
337  qErrnoWarning("index range over!");
338  return false;
339  }
340 
341  m_currentIndex = currentIndex;
342 
343  Q_FOREACH (QToolButton *button, m_tabList) {
344  button->setEnabled(true);
345  }
346 
347  at(currentIndex)->setFocus();
348  at(currentIndex)->setEnabled(false);
349 
350  updateHighlightGeometry();
351 
352  Q_EMIT currentChanged(currentIndex);
353  Q_EMIT currentTitleChanged(at(currentIndex)->text());
354 
355  return true;
356 }

398 bool DSegmentedControl::eventFilter(QObject *obj, QEvent *e)
399 {
400  if(m_currentIndex < 0)
401  return false;
402 
403  QWidget *w = at(m_currentIndex);
404 
405  if(obj == m_highlight){
406  if(e->type() == QEvent::Move){
407  updateHighlightGeometry(false);
408  }
409  }else if(obj == w){
410  if(e->type() == QEvent::Resize){
411  updateHighlightGeometry(false);
412  }
413  }
414 
415  return false;
416 }

418 void DSegmentedControl::updateHighlightGeometry(bool animation)
419 {
420  if(m_currentIndex<0)
421  return;
422 
423  QRect tmp = at(m_currentIndex)->geometry();
424 
425  if(m_currentIndex==0){
426  tmp.setX(0);
427  tmp.setWidth(tmp.width()+1);
428  }else if(m_currentIndex == count()-1){
429  tmp.setWidth(tmp.width()+1);
430  }
431  tmp.setY(0);
432 
433  if(m_highlight->geometry() == tmp)
434  return;
435 
436  if(animation){
437  m_highlightMoveAnimation->setStartValue(m_highlight->geometry());
438  m_highlightMoveAnimation->setEndValue(tmp);
439  m_highlightMoveAnimation->start();
440  }else{
441  m_highlight->setFixedSize(tmp.size());
442  m_highlight->move(tmp.topLeft());
443  }
444 }
