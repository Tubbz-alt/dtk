65 QPixmap DPictureSequenceViewPrivate::loadPixmap(const QString &path)
66 {
67  D_Q(DPictureSequenceView);
68 
69  qreal ratio = 1.0;
70 
71  const qreal devicePixelRatio = q->devicePixelRatioF();
72 
73  QPixmap pixmap;
74 
75  if (!qFuzzyCompare(ratio, devicePixelRatio)) {
76  QImageReader reader;
77  reader.setFileName(qt_findAtNxFile(path, devicePixelRatio, &ratio));
78  if (reader.canRead()) {
79  reader.setScaledSize(reader.size() * (devicePixelRatio / ratio));
80  pixmap = QPixmap::fromImage(reader.read());
81  pixmap.setDevicePixelRatio(devicePixelRatio);
82  }
83  } else {
84  pixmap.load(path);
85  }
86 
87  return pixmap;
88 }

90 void DPictureSequenceViewPrivate::_q_refreshPicture()
91 {
92  QGraphicsPixmapItem *item = pictureItemList.value(lastItemPos++);
93 
94  if (item)
95  item->hide();
96 
97  if (lastItemPos == pictureItemList.count()) {
98  lastItemPos = 0;
99 
100  if (singleShot)
101  refreshTimer->stop();
102 
103  D_QC(DPictureSequenceView);
104 
105  Q_EMIT q->playEnd();
106  }
107 
108  item = pictureItemList.value(lastItemPos);
109 
110  if (item)
111  item->show();
112 }

181 void DPictureSequenceView::setPictureSequence(const QList<QPixmap> &sequence, const bool autoScale)
182 {
183  D_D(DPictureSequenceView);
184 
185  stop();
186  d->scene->clear();
187  d->pictureItemList.clear();
188 
189  for (QPixmap pixmap : sequence) {
190  if (autoScale) {
191  pixmap = pixmap.scaled(size(), Qt::KeepAspectRatio, Qt::SmoothTransformation);
192  }
193 
194  d->pictureItemList << d->scene->addPixmap(pixmap);
195  d->pictureItemList.last()->hide();
196  }
197 
198  if (!d->pictureItemList.isEmpty()) {
199  d->pictureItemList.first()->show();
200  }
201 
202  setStyleSheet("background-color:transparent;");
203 }
