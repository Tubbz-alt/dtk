47 void DSpinBoxPrivate::init()
48 {
49  D_Q(DSpinBox);
50 
51  insideFrame = new QFrame(q);
52  insideFrame->setObjectName("SpinBoxInsideFrame");
53 
54  QHBoxLayout *h_layout = new QHBoxLayout(insideFrame);
55 
56  DImageButton *button_reset = DSpinBox_getButton("restore");
57  DImageButton *button_add = DSpinBox_getButton("spinner_increase");
58  DImageButton *button_sub = DSpinBox_getButton("spinner_decrease");
59 
60  h_layout->setMargin(0);
61  h_layout->setSpacing(0);
62  h_layout->addStretch();
63  h_layout->addWidget(button_reset);
64  h_layout->addWidget(button_add);
65  h_layout->addWidget(button_sub);
66 
67  q->connect(button_reset, &DImageButton::clicked, q, [q, this] {
68  if(defaultValue <= q->maximum() && defaultValue >= q->minimum())
69  q->setValue(defaultValue);
70  else
71  q->setValue(q->minimum());
72  });
73  q->connect(button_add, &DImageButton::clicked, q, &DSpinBox::stepUp);
74  q->connect(button_sub, &DImageButton::clicked, q, &DSpinBox::stepDown);
75 }
