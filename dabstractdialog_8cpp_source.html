448 void DAbstractDialog::resizeEvent(QResizeEvent *event)
449 {
450  if (event->size().width() >= maximumWidth()) {
451  bool resized = testAttribute(Qt::WA_Resized);
452 
453  setFixedWidth(maximumWidth());
454 
455  if (!resized)
456  setAttribute(Qt::WA_Resized, false);
457  }
458  QDialog::resizeEvent(event);
459 
460  D_DC(DAbstractDialog);
461 
462  if (!d->mouseMoved) {
463  setDisplayPosition(displayPosition());
464  }
465 
466  if (d->bgBlurWidget)
467  d->bgBlurWidget->resize(event->size());
468 
469  Q_EMIT sizeChanged(event->size());
470 }
