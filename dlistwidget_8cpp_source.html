160 void DListWidget::insertWidget(int index, QWidget *w, Qt::Alignment a)
161 {
162  Q_D(DListWidget);
163 
164  if(w==NULL || d->widgetList.contains(w))
165  return;
166 
167  if(d->itemWidth>0){
168  w->setFixedWidth(d->itemWidth);
169  }
170 
171  if(d->itemHeight>0){
172  w->setFixedHeight(d->itemHeight);
173  }
174 
175  if(index < 0) {
176  d->widgetList << w;
177  d->mainWidget->layout()->addWidget(w, 0, a);
178  } else {
179  d->widgetList.insert(index, w);
180  d->mainWidget->layout()->insertWidget(index, w, 0, a);
181  }
182 
183  d->mapVisible[w] = true;
184 
185  setVisibleCount(d->visibleCount + 1);
186 
187  Q_EMIT countChanged(count());
188 }

331 void DListWidget::setChecked(int index, bool checked)
332 {
333  Q_D(DListWidget);
334 
335  if (!d->checkable || (checked && isChecked(index)) || (!checked && !isChecked(index)))
336  return;
337 
338  if(d->checkMode == Radio && checked){
339  int checkedWidget = firstChecked();
340  if(checkedWidget>=0){
341  getWidget(checkedWidget)->setProperty("checked", false);
342  d->checkedList.removeOne(checkedWidget);
343  }
344  }
345 
346  if(index>=0 && index < count()){
347  QWidget *w = getWidget(index);
348  w->setProperty("checked", checked);
349  }
350 
351  if(checked){
352  d->checkedList << index;
353  }else{
354  d->checkedList.removeOne(index);
355  }
356 
357  if(d->checkMode == Radio && checked){
358  Q_EMIT firstCheckedChanged(index);
359  }
360 
361  Q_EMIT checkedChanged(index, checked);
362 }

637 {

644  QWidget *widget = childAt(e->pos());
